def FilteredSawtoothInstrument:
    param color number;
    param note_on boolean;

    osc TriSawOsc;
begin
    @osc(
        //color: mul(color, 0.5),
        color: 0.5,
        sample_rate: 44100,
        freq: 440
    )
end

def OuterInstrument:
    instr FilteredSawtoothInstrument;
begin
    @instr(color: 0.7, note_on: true)
end
