def Instrument:
    param sample_rate number;
    param freq number;
    param color number;
    param note_on boolean;

    osc TriSawOsc;
begin
    @osc(
        color: * $color 2,
        sample_rate: 44100,
        freq: $freq
    )
end

def OuterInstrument:
    instr Instrument;
begin
    @instr(sample_rate: 44100, freq: 440, color: 0.7, note_on: true)
end
