def Instrument:
    param sample_rate number;
    param freq number;
    param color number;
    param note_on boolean;

    osc PulseOsc;
begin
    @osc(
        color: $color,
        sample_rate: 44100,
        freq: * $freq 2
    )
end

def OuterInstrument:
    instr Instrument;
begin
    @instr(sample_rate: 44100, freq: 440, color: 0.7, note_on: true)
end
