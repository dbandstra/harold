// passed implicitly to child modules.

def Instrument:
    param sample_rate constant;
    param freq waveform;
    param color constant;
    param note_on boolean;

    osc PulseOsc;
    mod SineOsc;
    dist Distortion;
begin
    @dist(
        ingain: 0.25,
        outgain: 0.5,
        offset: 0,
        distortion_type: 'overdrive',
        input:
            * 0.7 @osc(
                color: $color,
                sample_rate: $sample_rate,
                freq:
                    * $freq
                        + 1
                            * 0.02 @mod(sample_rate: $sample_rate, freq: 4, phase: 0)
            )
    )
end

def OuterInstrument:
    param sample_rate constant;

    instr Instrument;
begin
    @instr(sample_rate: $sample_rate, freq: + 220 220, color: 0.7, note_on: true)
end
